Question8: "For each patient, compute the average PatientWeight across all their studies."

SELECT ?patient ?averagePatientWeight
WHERE {
  ?study rdf:type dcm:Study . 
  ?study dcm:PatientWeight ?patientWeight . 
  ?study dcm:Patient ?patient . 
  { ?study1 rdf:type dcm:Study . ?study1 dcm:Patient ?patient . } 
  OPTIONAL { ?study1 dcm:PatientWeight ?weight . } 
  BIND(AVG(?patientWeight) AS ?averagePatientWeight) 
} 
GROUP BY ?patient
ORDER BY ?patient
