Question8: "For each patient, compute the average PatientWeight across all their studies."

PREFIX dcm: <http://semantic-dicom.org/dcm#>
SELECT ?patient ?avgWeight
WHERE {
  ?study dcm:PatientID ?patient .
  ?study dcm:PatientWeight ?weight .
  BIND(AVG(?weight) AS ?avgWeight)
}
GROUP BY ?patient
ORDER BY ?patient

  ?study dcm:Modality ?modality .
  ?patient dcm:PatientsAge ?age .
  ?patient dcm:PatientSex ?sex .

Question40: "What is the date difference between the earliest and latest study for Patient 'David Green'?"

PREFIX dcm: <http://semantic-dicom.org/dcm#>
SELECT ?earliestDate ?latestDate (xsd:date(?latestDate) - xsd:date(?earliestDate)) AS ?dateDiff
WHERE {
  ?study dcm:PatientID "David Green" .
  ?study dcm:StudyDate ?studyDate .
  BIND(MIN(?studyDate) AS ?earliestDate)
  BIND(MAX(?studyDate) AS ?latestDate)
}
ORDER BY ?dateDiff DESC
LIMIT 1
