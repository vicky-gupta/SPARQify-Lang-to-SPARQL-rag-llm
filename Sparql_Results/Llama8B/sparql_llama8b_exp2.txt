Question1: "For the patient with PatientID “P24680”, show their PatientSize and PatientWeight."

SELECT?patientSize?patientWeight WHERE {?study rdf:type dcm:Study.?study dcm:PatientID "P24680".?study dcm:PatientSize?patientSize.?study dcm:PatientWeight?patientWeight. }





Question2: "Which studies UID have RetrieveAETitle equal to “CT_ARCHIVE_A”?"

SELECT?studyInstanceUID WHERE {?study rdf:type dcm:Study.?study dcm:RetrieveAETitle "CT_ARCHIVE_A".?study dcm:StudyInstanceUID?studyInstanceUID } LIMIT 100





Question3: "Find all seriesUID of images with modality CT performed by PerformingPhysiciansName “Dr. Johnson” between 08:00:00 and 12:00:00."

SELECT?seriesUID WHERE {?image rdf:type dcm:Image.?image dcm:Modality?modality.?image dcm:SeriesInstanceUID?seriesUID.?image dcm:PerformedBy?performingPhysician. FILTER(?modality = "CT" && CONTAINS(STR(?performingPhysician), "Johnson") && STRSTARTS(STR(?performingPhysician), "Dr.") && STR(?performingPhysician) @ EN("08:00:00") @ ET("12:00:00")) } LIMIT 100

Question43: "Calculate the minimum and maximum Rescale Intercept among all CT images"

SELECT?minRescaleIntercept?maxRescaleIntercept
WHERE {
 ?image rdf:type dcm:Image.
 ?image dcm:Modality?modality.
  FILTER(?modality = "CT")
  OPTIONAL {?image dcm:RescaleIntercept?rescaleIntercept }
}
GROUP BY?image
HAVING (MIN(?rescaleIntercept) AS?minRescaleIntercept)
HAVING (MAX(?rescaleIntercept) AS?maxRescaleIntercept)
